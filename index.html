  • Otwórz Streamer.bot → zakładka Actions

  • W każdej akcji dodaj sub-action: Core → C# → Execute Code i wklej odpowiedni kod z sekcji poniżej
  • Dla SubGoal Subscribe: dodaj trigger Twitch → Subscribe i Twitch → Re-Subscribe
  • Dla SubGoal Gift: dodaj trigger Twitch → Gift Subscription
  • Dla SubGoal Sync: jako pierwszy sub-action dodaj Twitch → Subscribers → Get Subscriber Count, potem Execute Code
  • Upewnij się że WebSocket Server jest włączony: Servers/Clients → WebSocket Server → Auto Start ✓ (port 8080)

3 Konfiguracja OBS

  • Dodaj nowe źródło: Browser Source, nazwij je Sub Goal
  • W polu URL wpisz ścieżkę do pliku: file:///C:/subgoal/subgoal-bar.html (dostosuj ścieżkę)

  • Zaznacz opcję „Allow OBS to control this browser source" – bez tego pasek nie odbierze eventów!

3 OBS Setup

  • Add a new source: Browser Source, name it Sub Goal
  • Set the URL to the local file path: file:///C:/subgoal/subgoal-bar.html (adjust as needed)
  • Set width and height to match bar orientation (e.g. 800×60 for horizontal)
  • Check „Allow OBS to control this browser source" – without this the bar won't receive events!
  • Click Refresh after every HTML file change

C# SubGoal Subscribe — Trigger: Twitch → Subscribe + Re-Subscribe 
using System;

// AKCJA: SubGoal Subscribe
// Trigger: Twitch → Subscribe  +  Twitch → Re-Subscribe

public class CPHInline
{
    private const string OBS_SOURCE = "Sub Goal";
    private const int    OBS_CONN   = 0;

    public bool Execute()
    {
        string username = "";
        CPH.TryGetArg("userName", out username);
        if (string.IsNullOrEmpty(username)) username = "Unknown";

        string safeUser = username
            .Replace("\\", "\\\\")
            .Replace("\"", "\\\"");

        string eventData = "{\"source\":\"subgoal-panel\",\"type\":\"addSub\","
                         + "\"data\":{\"username\":\"" + safeUser + "\","
                         + "\"subCount\":-1}}";

        string payload = "{\"vendorName\":\"obs-browser\","
                       + "\"requestType\":\"emit_event\","
                       + "\"requestData\":{"
                       + "\"event_name\":\"obsCustomEvent\","
                       + "\"event_data\":" + eventData + "}}";

        CPH.ObsSendRaw("CallVendorRequest", payload, OBS_CONN);
        CPH.LogInfo("[SubGoal] Nowy sub: " + username);

        // Po 2s pobierz dokładną liczbę z Twitch API i zaktualizuj pasek + panel
        CPH.Wait(2000);
        CPH.RunAction("SubGoal Sync", false);
        return true;
    }
}
C# SubGoal Gift — Trigger: Twitch → Gift Subscription 
using System;

// AKCJA: SubGoal Gift
// Trigger: Twitch → Gift Subscription

public class CPHInline
{
    private const int OBS_CONN = 0;

    public bool Execute()
    {
        string gifter = "";
        CPH.TryGetArg("userName", out gifter);
        if (string.IsNullOrEmpty(gifter)) gifter = "Anonim";

        int amount = 1;
        CPH.TryGetArg("amount", out amount);
        if (amount < 1) amount = 1;

        string safeGifter = gifter
            .Replace("\\", "\\\\")
            .Replace("\"", "\\\"");

        for (int i = 0; i < amount; i++)
        {
            string label = amount > 1
                ? safeGifter + " [gift " + (i + 1) + "/" + amount + "]"
                : safeGifter + " [gift]";

            string eventData = "{\"source\":\"subgoal-panel\",\"type\":\"addSub\","
                             + "\"data\":{\"username\":\"" + label + "\","
                             + "\"subCount\":-1}}";

            string payload = "{\"vendorName\":\"obs-browser\","
                           + "\"requestType\":\"emit_event\","
                           + "\"requestData\":{"
                           + "\"event_name\":\"obsCustomEvent\","
                           + "\"event_data\":" + eventData + "}}";

            CPH.ObsSendRaw("CallVendorRequest", payload, OBS_CONN);
            if (i < amount - 1) CPH.Wait(300);
        }

        CPH.LogInfo("[SubGoal] Gift sub: " + gifter + " x" + amount);

        // Sync po zakończeniu animacji
        CPH.Wait(amount * 300 + 500);
        CPH.RunAction("SubGoal Sync", false);
        return true;
    }
}
C# SubGoal Sync — ⚠️ Wymagany 1. sub-action: Get Subscriber Count
⚠️ W Streamer.bot jako pierwszy sub-action w tej akcji dodaj: Twitch → Subscribers → Get Subscriber Count — dopiero potem Execute Code z tym kodem. 
using System;

// AKCJA: SubGoal Sync
// Sub-actions kolejność:
//   1. Twitch → Subscribers → Get Subscriber Count
//   2. Execute Code (ten kod)

public class CPHInline
{
    private const int OBS_CONN = 0;

    public bool Execute()
    {
        int subCount = 0;
        CPH.TryGetArg("subscriberCount", out subCount);

        string barEvent = "{\"source\":\"subgoal-panel\",\"type\":\"setSubs\","
                        + "\"data\":{\"value\":" + subCount + "}}";

        string obsPayload = "{\"vendorName\":\"obs-browser\","
                          + "\"requestType\":\"emit_event\","
                          + "\"requestData\":{"
                          + "\"event_name\":\"obsCustomEvent\","
                          + "\"event_data\":" + barEvent + "}}";

        CPH.ObsSendRaw("CallVendorRequest", obsPayload, OBS_CONN);
        CPH.SetGlobalVar("subgoal_current", subCount, true);

        string broadcastJson = "{\"type\":\"subgoal-sync\",\"subCount\":" + subCount + "}";
        CPH.WebsocketBroadcastJson(broadcastJson);

        CPH.LogInfo("[SubGoal Sync] subCount=" + subCount);
        return true;
    }
}
C# SubGoal Send — Relay eventów z control-panel do OBS 
using System;
using Newtonsoft.Json.Linq;

// AKCJA: SubGoal Send
// Wywoływana przez control-panel.html przez WebSocket

public class CPHInline
{
    private const string OBS_SOURCE = "Sub Goal";
    private const int    OBS_CONN   = 0;

    public bool Execute()
    {
        string eventJson = "";
        CPH.TryGetArg("subgoalEventJson", out eventJson);

        if (string.IsNullOrEmpty(eventJson))
        {
            CPH.LogWarn("[SubGoal Send] Brak argumentu subgoalEventJson");
            return false;
        }

        try { JToken.Parse(eventJson); }
        catch (Exception ex)
        {
            CPH.LogWarn("[SubGoal Send] Nieprawidłowy JSON: " + ex.Message);
            return false;
        }

        string payload = "{\"vendorName\":\"obs-browser\","
                       + "\"requestType\":\"emit_event\","
                       + "\"requestData\":{"
                       + "\"event_name\":\"obsCustomEvent\","
                       + "\"event_data\":" + eventJson + "}}";

        CPH.ObsSendRaw("CallVendorRequest", payload, OBS_CONN);
        CPH.LogInfo("[SubGoal Send] Wysłano: " + eventJson);
        return true;
    }
}
